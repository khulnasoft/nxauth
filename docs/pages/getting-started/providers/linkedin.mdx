### Configuration

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "nxauth"
import LinkedIn from "nxauth/providers/linkedin"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [LinkedIn],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@nxauth.ts"
import { QwikAuth$ } from "@nxauth/qwik"
import LinkedIn from "@nxauth/qwik/providers/linkedin"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [LinkedIn],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@nxauth/sveltekit"
import LinkedIn from "@nxauth/sveltekit/providers/linkedin"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [LinkedIn],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@nxauth/express"
import LinkedIn from "@nxauth/express/providers/linkedin"

app.use("/auth/*", ExpressAuth({ providers: [LinkedIn] }))
```

  </Code.Express>
