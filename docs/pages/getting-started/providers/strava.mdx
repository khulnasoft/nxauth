### Configuration

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "nxauth"
import Strava from "nxauth/providers/strava"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [Strava],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@nxauth.ts"
import { QwikAuth$ } from "@nxauth/qwik"
import Strava from "@nxauth/qwik/providers/strava"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [Strava],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@nxauth/sveltekit"
import Strava from "@nxauth/sveltekit/providers/strava"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [Strava],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@nxauth/express"
import Strava from "@nxauth/express/providers/strava"

app.use("/auth/*", ExpressAuth({ providers: [Strava] }))
```

  </Code.Express>
