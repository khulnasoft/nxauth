### Environment Variables

```
AUTH_TWITCH_ID
AUTH_TWITCH_SECRET
```

### Configuration

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "nxauth"
import Twitch from "nxauth/providers/twitch"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [Twitch],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@nxauth.ts"
import { QwikAuth$ } from "@nxauth/qwik"
import Twitch from "@nxauth/qwik/providers/twitch"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [Twitch],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@nxauth/sveltekit"
import Twitch from "@nxauth/sveltekit/providers/twitch"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [Twitch],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@nxauth/express"
import Twitch from "@nxauth/express/providers/twitch"

app.use("/auth/*", ExpressAuth({ providers: [Twitch] }))
```

  </Code.Express>
</Code>

### Notes

- Twitch will redirect to the first redirect URI if multiple are added.
