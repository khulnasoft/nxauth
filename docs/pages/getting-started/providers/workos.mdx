### Environment Variables

```
AUTH_WORKOS_ID
AUTH_WORKOS_SECRET
```

WorkOS also requires you to pass in your `connection` ID to the provider.

### Configuration

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "nxauth"
import WorkOS from "nxauth/providers/workos"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [WorkOS({ connection: "conn_abc123" })],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@nxauth.ts"
import { QwikAuth$ } from "@nxauth/qwik"
import WorkOS from "@nxauth/qwik/providers/workos"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [WorkOS({ connection: "conn_abc123" })],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@nxauth/sveltekit"
import WorkOS from "@nxauth/sveltekit/providers/workos"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [WorkOS({ connection: "conn_abc123" })],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@nxauth/express"
import WorkOS from "@nxauth/express/providers/workos"

app.use(
  "/auth/*",
  ExpressAuth({ providers: [WorkOS({ connection: "conn_abc123" })] })
)
```
